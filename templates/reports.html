🏆 Student Ranking Using RANK() DENSE_RANK() ROW_NUMBER() 
SELECT name, total_marks,
    RANK()       OVER (ORDER BY total_marks DESC) AS rnk,
    DENSE_RANK() OVER (ORDER BY total_marks DESC) AS dense_rnk,
    ROW_NUMBER() OVER (ORDER BY total_marks DESC) AS row_num
FROM vw_student_performance

💡 Difference Explained:
  • RANK() — Leaves gaps after ties (1, 1, 3, 4)
  • DENSE_RANK() — No gaps after ties (1, 1, 2, 3)
  • ROW_NUMBER() — Always unique number (1, 2, 3, 4)
⭐ Top Students (CTE) Students with avg marks > 70 
WITH top_students AS (
    SELECT s.name,
           ROUND(AVG(m.marks), 2) AS avg_m
    FROM students s
    JOIN marks m ON s.student_id = m.student_id
    GROUP BY s.student_id, s.name
)
SELECT * FROM top_students
WHERE avg_m > 70
ORDER BY avg_m DESC

    SELECT student_id
    FROM marks
    GROUP BY student_id
    HAVING AVG(marks) > 70
